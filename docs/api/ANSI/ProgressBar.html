def initialize(title, total, out=STDERR)
  @title = title
  @total = total
  @out   = out

  @bar_length = 80
  @bar_mark = "o"

def set(count)
  if count < 0
    raise "invalid count less than zero: #{count}"
  elsif count > @total
    if @total_overflow
      @total = count + 1
    else
      raise "invalid count greater than total: #{count}"
    end
  end
  @current = count
  show_progress
  @previous = @current
end
